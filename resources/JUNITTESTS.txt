public class Login {
    private String email;
    private String password;
    private int studentID;

    public Login(String email, String password, int studentID) {
        this.email = email;
        this.password = password;
        this.studentID = studentID;
    }

    public boolean checkLogin(String email, String password) {
        return this.email.equals(email) && this.password.equals(password);
    }

    public boolean isValidStudentID(int studentID) {
        return studentID >= 10000000 && studentID <= 99999999;
    }
}

import java.util.ArrayList;
import java.util.List;

public class EventManager {

    private List<Event> events = new ArrayList<>();

    public void addEvent(Event event) {
        events.add(event);
    }

    public void deleteEvent(String eventName) {
        events.removeIf(event -> event.getEventName().equals(eventName));
    }

    public void updateEvent(String eventName, String newDescription,
                          String newLocation, String newDate, String newTime) {

        for (Event event : events) {
            if (event.getEventName().equals(eventName)) {

                event.setDescription(newDescription);
                event.setLocation(newLocation);
                event.setDate(newDate);
                event.setTime(newTime);

                return;
            }
        }
    }

    public List<Event> getEventList() {
        return events;
    }
}

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;
class EventTest {
    private EventManager eventManager;

    @BeforeEach
    void setUp() {
        eventManager = new EventManager();
        eventManager.addEvent(new Event("AP Cryptography", "Lab 9", "D04", "Monday 13th April", "3:15PM"));
        eventManager.addEvent(new Event("Women In Tech Meeting", "Speaker + Info Session", "F09", "Monday 20th April", "5:15PM"));
        eventManager.addEvent(new Event("Counseling", "Video Call on Zoom", "Online", "Thursday 16th April", "11:30AM"));
    }

    @Test
    void testAddEvent() {
        eventManager.addEvent(new Event("Software Engineering Practice", "Lab - Working on CA!", "IT221", "Tuesday 14th April", "2:15PM"));
        assertEquals(4, eventManager.getEventList().size());
    }

    @Test
    void testDeleteEvent() {
        eventManager.deleteEvent("Women In Tech Meeting");
        assertEquals(2, eventManager.getEventList().size());
    }

    @Test
    void testUpdateEventChange() {
        eventManager.updateEvent("Women In Tech Meeting", "Cancelled! Online Committee meeting soon", "Online", "Thursday 23rd April", "8:00PM");
        Event event = eventManager.getEventList().stream()
                .filter(e -> e.getEventName().equals("Women In Tech Meeting"))
                .findFirst()
                .orElse(null);
        if (event != null) {
            assertEquals("Women In Tech Meeting", event.getEventName());
            assertEquals("Cancelled! Online Committee meeting soon", event.getDescription());
            assertEquals("Online", event.getLocation());
            assertEquals("Thursday 23rd April", event.getDate());
            assertEquals("8:00PM", event.getTime());
        }
    }

    @Test
    void testUpdateEventNotEqualNull() {
        eventManager.updateEvent("Women In Tech Meeting", "Cancelled! Online Committee meeting soon", "Online", "Thursday 23rd April", "8:00PM");
        Event event = eventManager.getEventList().stream()
                .filter(e -> e.getEventName().equals("Women In Tech Meeting"))
                .findFirst()
                .orElse(null);
        assertNotNull(event);
    }

}
